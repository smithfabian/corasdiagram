\newcommand{\AuthorTableRows}{}     % rows for the cover table
\makeatletter
\let\PdfAuthorString\@empty
\makeatother

\makeatletter
\newcommand{\AddAuthor}[2]{%
  % (Optional) keep structured storage if you want it elsewhere
  \g@addto@macro\AuthorList{\AuthorEntry{#1}{#2}}%
  % Build cover table rows directly (no redef tricks later)
  \g@addto@macro\AuthorTableRows{#1 & #2\tabularnewline}%
  % Build PDF metadata string
  \ifx\PdfAuthorString\@empty
    \gdef\PdfAuthorString{#1}%
  \else
    \g@addto@macro\PdfAuthorString{; #1}%
  \fi
}
\makeatother
