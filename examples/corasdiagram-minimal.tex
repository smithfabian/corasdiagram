% Minimal semantic-first CORAS example.
% Compile from the repository root with:
%   TEXINPUTS=tex/latex//: pdflatex -interaction=nonstopmode -halt-on-error \
%     examples/corasdiagram-minimal.tex
\documentclass{article}
\usepackage[margin=20mm]{geometry}
\usepackage{corasdiagram}

\begin{document}

\begin{center}
\begin{corasassetdiagram}[x=1cm,y=1cm,asset columns=3]
  \party[id=company]{Company}
  \asset(direct)[id=availability]{Availability of server}
  \asset(direct)[id=confidentiality,order=2]{Confidentiality of information}
  \asset(indirect)[id=reputation,order=3]{Company's reputation}

  \assigns{company -> availability}[asset value=critical]
  \assigns{company -> confidentiality}[asset value=critical]
  \assigns{company -> reputation}
  \harms{availability -> reputation}
  \harms{confidentiality -> reputation}
\end{corasassetdiagram}
\end{center}

\end{document}
