\section*{Typed Threat Branch Regression}

\begin{center}
\begin{corasthreatdiagram}[x=.78cm,y=.78cm,min edge gap=8mm]
  \lane{1}{
    \threat(deliberate)[id=threat-delib-hacker-malware]{Hacker}
    \vulnerability[id=vuln-unpatched-system]{Unpatched\\system}
    \vulnerability[id=vuln-poor-security-design]{Poor security\\design}
    \scenario[id=scenario-unauthorized-access]{Unauthorised\\access to the\\system}
    \vulnerability[id=vuln-easy-admin-permission]{Easy way to get\\admin permission}
    \vulnerability[id=vuln-sensor-integrity-not-checked]{Not checking the integrity\\of sensor data}
    \scenario[id=scenario-malware-in-ecdis]{Malware in ECDIS\\system}
    \scenario[id=scenario-sensor-data-manipulation]{Sensor data\\manipulation}
    \incident[id=ui-ecdis-unavailable]{ECDIS system\\not available}
  }

  \lane{2}{
    \threat(nonhuman)[id=threat-nonhuman-hardware]{Non Human Threat}
    \vulnerability[id=vuln-substandard-materials]{Substandard materials used\\to produce hardware}
    \scenario[id=scenario-hardware-error]{Hardware error}
    \incident[id=ui-ecdis-degraded]{ECDIS system\\degraded}
  }

  \lane{3}{
    \threat(accidental)[id=threat-accidental-crew-member]{Crew member}
    \vulnerability[id=vuln-overworked-or-untrained-staff]{Overworked staff/staff that\\lacks knowledge}
    \scenario[id=scenario-human-error]{Human error}
    \incident[id=ui-ecdis-false-data]{ECDIS system\\displaying false data}
  }

  \asset[id=asset-ecdis-integrity-and-availability,right of=ui-ecdis-degraded]{ECDIS integrity\\and availability}

  \cause{threat-delib-hacker-malware -> scenario-unauthorized-access}[via={vuln-unpatched-system,vuln-poor-security-design}]
  \cause{scenario-unauthorized-access -> scenario-malware-in-ecdis}[via={vuln-easy-admin-permission}]
  \cause{scenario-unauthorized-access -> scenario-sensor-data-manipulation}[via={vuln-sensor-integrity-not-checked}]
  \cause{threat-nonhuman-hardware -> scenario-hardware-error}[via={vuln-substandard-materials}]
  \cause{threat-accidental-crew-member -> scenario-human-error}[via={vuln-overworked-or-untrained-staff}]

  \cause{{scenario-malware-in-ecdis,scenario-sensor-data-manipulation,scenario-hardware-error} -> ui-ecdis-unavailable}
  \cause{{scenario-malware-in-ecdis,scenario-sensor-data-manipulation,scenario-hardware-error,scenario-human-error} -> ui-ecdis-degraded}
  \cause{{scenario-sensor-data-manipulation,scenario-hardware-error,scenario-human-error} -> ui-ecdis-false-data}

  \impact{ui-ecdis-unavailable -> asset-ecdis-integrity-and-availability}
  \impact{ui-ecdis-degraded -> asset-ecdis-integrity-and-availability}
  \impact{ui-ecdis-false-data -> asset-ecdis-integrity-and-availability}
\end{corasthreatdiagram}
\end{center}
