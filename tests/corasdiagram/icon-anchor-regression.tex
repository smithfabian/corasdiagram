\newcommand{\corastestdrawanchorgrid}[1]{%
  \corasdiagram@getbordernode{\corastest@bordernode}{#1}%
  \begin{scope}[shift={(\corastest@bordernode.center)}]
    \draw[step=1,draw=black!10,line width=0.08pt] (-40,-28) grid (40,28);
    \draw[step=5,draw=black!18,line width=0.12pt] (-40,-28) grid (40,28);
    \draw[draw=black!28,line width=0.18pt] (-40,0) -- (40,0);
    \draw[draw=black!28,line width=0.18pt] (0,-28) -- (0,28);
    \foreach \x in {-40,-35,...,40} {
      \node[
        font=\ttfamily\tiny,
        text=black!45,
        fill=white,
        fill opacity=0.7,
        text opacity=1,
        inner sep=0.2pt,
        outer sep=0pt,
        anchor=north
      ] at (\x,-1.4) {\x};
    }
    \foreach \y in {-25,-20,...,25} {
      \node[
        font=\ttfamily\tiny,
        text=black!45,
        fill=white,
        fill opacity=0.7,
        text opacity=1,
        inner sep=0.2pt,
        outer sep=0pt,
        anchor=east
      ] at (-1.2,\y) {\y};
    }
    \node[
      font=\ttfamily\tiny,
      text=black!55,
      fill=white,
      fill opacity=0.8,
      text opacity=1,
      inner sep=0.25pt,
      outer sep=0pt,
      anchor=west
    ] at (41,0) {x / mm};
    \node[
      font=\ttfamily\tiny,
      text=black!55,
      fill=white,
      fill opacity=0.8,
      text opacity=1,
      inner sep=0.25pt,
      outer sep=0pt,
      anchor=south
    ] at (0,29) {y / mm};
  \end{scope}%
}

\newcommand{\corastestdrawiconanchor}[5]{%
  \corasdiagram@getbordernode{\corastest@bordernode}{#1}%
  \node[
    font=\ttfamily\scriptsize,
    fill=white,
    fill opacity=0.92,
    text opacity=1,
    inner sep=0.5pt,
    outer sep=0pt
  ] at ($(\corastest@bordernode.center)+(#3,#4)$) {#2};
  \draw[coras/relation]
    ($(\corastest@bordernode.center)+(#3,#4)$)
    -- (\corastest@bordernode.#5);
}

\newcommand{\corastestformatdim}[2]{%
  \pgfmathparse{round((\strip@pt\dimexpr#1\relax/2.84527559)*100)/100}%
  \xdef#2{\pgfmathresult}%
}

\newcommand{\corastestlegendrow}[3]{%
  \corasdiagram@getanchorlegendcoord{#1}{#3}{#2}{\corastest@legendx}{\corastest@legendy}%
  \corastestformatdim{\corastest@legendx}{\corastest@legendxtext}%
  \corastestformatdim{\corastest@legendy}{\corastest@legendytext}%
  #2 & (\corastest@legendxtext,\corastest@legendytext)\,mm\\%
}

\newcommand{\corastestdrawiconanchorfan}[1]{%
  \corastestdrawiconanchor{#1}{north west}{-22}{19}{north west}%
  \corastestdrawiconanchor{#1}{north}{0}{26}{north}%
  \corastestdrawiconanchor{#1}{north east}{22}{19}{north east}%
  \corastestdrawiconanchor{#1}{west}{-28}{4}{west}%
  \corastestdrawiconanchor{#1}{east}{28}{4}{east}%
  \corastestdrawiconanchor{#1}{south west}{-22}{-19}{south west}%
  \corastestdrawiconanchor{#1}{south}{0}{-26}{south}%
  \corastestdrawiconanchor{#1}{south east}{22}{-19}{south east}%
}

\newcommand{\corastestdrawanchortable}[1]{%
  \corasdiagram@getbordernode{\corastest@bordernode}{#1}%
  \node[
    anchor=west,
    draw=black!45,
    fill=white,
    fill opacity=0.95,
    text opacity=1,
    rounded corners=0.6mm,
    line width=0.2pt,
    inner sep=1.1mm,
    outer sep=0pt
  ] at ($(\corastest@bordernode.center)+(52,0)$) {%
    \begingroup
    \ttfamily\tiny
    \renewcommand{\arraystretch}{0.92}%
    \begin{tabular}{@{}ll@{}}
    anchor & position (x,y)\\
    \corastestlegendrow{#1}{north west}{northwest}%
    \corastestlegendrow{#1}{north}{north}%
    \corastestlegendrow{#1}{north east}{northeast}%
    \corastestlegendrow{#1}{west}{west}%
    \corastestlegendrow{#1}{east}{east}%
    \corastestlegendrow{#1}{south west}{southwest}%
    \corastestlegendrow{#1}{south}{south}%
    \corastestlegendrow{#1}{south east}{southeast}%
    \end{tabular}%
    \endgroup
  };
}
