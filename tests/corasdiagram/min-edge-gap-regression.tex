\makeatletter
\newdimen\corastest@first
\newdimen\corastest@second
\newdimen\corastest@gap

\newcommand{\corastest@getborderx}[3]{%
  \corasdiagram@getnodeprop{bordernode}{#1}{\corastest@bordernode}%
  \pgf@process{%
    \pgfpointshapeborder{\corastest@bordernode}{\pgfpointanchor{#2}{center}}%
  }%
  #3=\pgf@x\relax
}

\newcommand{\corastest@getbordery}[3]{%
  \corasdiagram@getnodeprop{bordernode}{#1}{\corastest@bordernode}%
  \pgf@process{%
    \pgfpointshapeborder{\corastest@bordernode}{\pgfpointanchor{#2}{center}}%
  }%
  #3=\pgf@y\relax
}

\newcommand{\corastestassertminhorizontalgap}[3]{%
  \corastest@getborderx{#1}{#2}{\corastest@first}%
  \corastest@getborderx{#2}{#1}{\corastest@second}%
  \corastest@gap=\dimexpr\corastest@second-\corastest@first\relax
  \ifdim\corastest@gap<#3\relax
    \PackageError{corasdiagram-tests}{Horizontal gap between `#1' and `#2' is \the\corastest@gap, expected at least #3}{%
      Auto and relative placement should preserve the configured minimum edge gap.%
    }%
  \fi
}

\newcommand{\corastestassertminverticalgap}[3]{%
  \corastest@getbordery{#1}{#2}{\corastest@first}%
  \corastest@getbordery{#2}{#1}{\corastest@second}%
  \corastest@gap=\dimexpr\corastest@first-\corastest@second\relax
  \ifdim\corastest@gap<#3\relax
    \PackageError{corasdiagram-tests}{Vertical gap between `#1' and `#2' is \the\corastest@gap, expected at least #3}{%
      Auto and relative placement should preserve the configured minimum edge gap.%
    }%
  \fi
}

\newcommand{\corastestassertmaxhorizontalgap}[3]{%
  \corastest@getborderx{#1}{#2}{\corastest@first}%
  \corastest@getborderx{#2}{#1}{\corastest@second}%
  \corastest@gap=\dimexpr\corastest@second-\corastest@first\relax
  \ifdim\corastest@gap>#3\relax
    \PackageError{corasdiagram-tests}{Horizontal gap between `#1' and `#2' is \the\corastest@gap, expected at most #3}{%
      Absolute coordinates should bypass minimum edge gap enforcement.%
    }%
  \fi
}
\makeatother
